#include <Arduino.h>
#include "DomoBoard.h"

DomoBoard board;

volatile bool flag = false;

bool st3 = false;
bool st10 = false;
bool st11 = false;

int last10 = HIGH;
int last11 = HIGH;

void isr3() {
  flag = true;
}

void setup() {
  board.begin();
  attachInterrupt(digitalPinToInterrupt(board.P3), isr3, FALLING);
}

void loop() {
  if (flag) {
    delay(50);
    if (digitalRead(board.P3) == LOW) {
       st3 = !st3; 
       if (st3) Serial.println("Interruptor 2: ON");
       else Serial.println("Interruptor 2: OFF");
    }
    flag = false;
  }

  int curr10 = board.read10();
  if (curr10 != last10) {
      if (curr10 == LOW) { 
        st10 = !st10;
        if (st10) Serial.println("Interruptor 1: ON");
        else Serial.println("Interruptor 1: OFF");
      }
      last10 = curr10;
      delay(50);
  }

  int curr11 = board.read11();
  if (curr11 != last11) {
      if (curr11 == LOW) { 
        st11 = !st11;
        if (st11) Serial.println("Interruptor Opt: ON");
        else Serial.println("Interruptor Opt: OFF");
      }
      last11 = curr11;
      delay(50);
  }
}
