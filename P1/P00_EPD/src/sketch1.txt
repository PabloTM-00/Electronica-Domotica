#include <Arduino.h>
#include "DomoBoard.h"

DomoBoard board;

volatile bool flag = false;
int last10 = HIGH;
int last11 = HIGH;

void isr3() {
  flag = true;
}

void setup() {
  board.begin();
  attachInterrupt(digitalPinToInterrupt(board.P3), isr3, CHANGE);
}

void loop() {
  if (flag) {
    delay(50);
    if (digitalRead(board.P3) == HIGH) {
       Serial.println("Boton 2 -> Pulsado");
    } else {
       Serial.println("Boton 2 -> Soltado");
    }
    flag = false;
  }

  int curr10 = board.read10();
  if (curr10 != last10) {
      if (curr10 == HIGH) {
        Serial.println("Boton 1 -> Pulsado");
      } else {
        Serial.println("Boton 1 -> Soltado");
      }
      last10 = curr10;
      delay(50);
  }

  int curr11 = board.read11();
  if (curr11 != last11) {
      if (curr11 == LOW) {
        Serial.println("Opto 11 -> Activado");
      } else {
        Serial.println("Opto 11 -> Desactivado");
      }
      last11 = curr11;
      delay(50);
  }
}
